\section{Formulas}

\begin{itemize}
	\item Bayes' rule 
	\begin{eqn}
		P(y = 1 \mid x) = \frac{P(x \mid y = 1)P(y = 1)}{P(x)}
	\end{eqn}
\end{itemize}

\subsection{LDA}

\begin{eqn}
	P(y = 0) = \frac{N_0}{N_0 + N_1} \\
	\mu_0 = \sum_{i=1:n}{I(y_i = 0) \frac{x_i}{N_0}} \\
	\Sigma = \sum_{k = 0:1} \sum_{i = 1:n} \frac{I(y_i = 0)(x_i - \mu_k)(x_i - \mu_k)^T}{N_0 + N_1 - N_k}
\end{eqn}
